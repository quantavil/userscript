### 3. **Pack moves as 32-bit integers — eliminate GC pressure in the hottest path**

Every `createMove` allocates an object. Millions per search. Use a packed int:

```js
// Encoding: from(6) | to(6) | flags(3) | promo(3) | piece(4) | captured(4) = 26 bits
function encodeMove(from, to, flags, promo, piece, captured) {
    return (from) | (to << 6) | (flags << 12) | ((promo & 7) << 15)
         | (((piece + 6) & 0xF) << 18) | (((captured + 6) & 0xF) << 22);
}
function decodeFrom(mv)     { return mv & 0x3F; }
function decodeTo(mv)       { return (mv >> 6) & 0x3F; }
function decodeFlags(mv)    { return (mv >> 12) & 0x7; }
function decodePromo(mv)    { return ((mv >> 15) & 0x7) - 6; } // adjust back
function decodePiece(mv)    { return ((mv >> 18) & 0xF) - 6; }
function decodeCaptured(mv) { return ((mv >> 22) & 0xF) - 6; }
```

This eliminates **millions of object allocations** per search and enables storing moves in typed arrays.

### 4. **Add Late Move Reductions (LMR) — the single biggest search speedup you're missing**

```js
// In negamax, after the first few moves at non-root:
let score;
if (movesSearched === 0) {
    score = -this.negamax(depth - 1, -beta, -alpha, ply + 1, childPv, ext);
} else {
    let reduction = 0;
    // LMR: reduce late quiet moves
    if (depth >= 3 && movesSearched >= 3
        && mv.captured === EMPTY && !(mv.flags & FLAG_PROMO) && !inChk) {
        reduction = 1 + Math.log2(depth) * Math.log2(movesSearched) * 0.4 | 0;
        reduction = Math.min(reduction, depth - 2);
    }

    score = -this.negamax(depth - 1 - reduction, -alpha - 1, -alpha, ply + 1, childPv, ext);

    if (score > alpha && (reduction > 0 || score < beta)) {
        childPv.length = 0;
        score = -this.negamax(depth - 1, -beta, -alpha, ply + 1, childPv, ext);
    }
}
```

Typical speedup: **2–5× deeper** at equal time.

### 6. **Add basic SEE (Static Exchange Evaluation) for capture ordering and pruning**

```js
see(sq, sideToMove) {
    const gain = [];
    let side = sideToMove;
    let piece = this.smallestAttacker(sq, side);
    if (!piece) return 0;

    gain.push(PIECE_VAL[Math.abs(this.board[sq])]);
    let val = PIECE_VAL[Math.abs(piece)];

    // Simulate exchange sequence
    while (piece) {
        side = -side;
        gain.push(val - gain[gain.length - 1]);
        val = PIECE_VAL[Math.abs(piece)];
        // Remove attacker, find next smallest
        piece = this.smallestAttacker(sq, side); // excluding removed pieces
    }

    // Minimax the gain stack
    for (let i = gain.length - 2; i >= 0; i--) {
        gain[i] = -Math.max(-gain[i], gain[i + 1]);
    }
    return gain[0];
}
```

Use in move ordering (`scoreMoves`) and to **prune losing captures** in quiescence.

### 7. **Fix repetition detection — avoid string allocation on every node**

```js
// CURRENT (allocates string every makeMove):
this.positionHistory.push(this.hash[0] + '|' + this.hash[1]);

// FIX — store raw pairs, compare numerically:
this.posHistoryHi = new Int32Array(1024);
this.posHistoryLo = new Int32Array(1024);
this.posHistoryLen = 0;

// In makeMove:
this.posHistoryHi[this.posHistoryLen] = this.hash[0];
this.posHistoryLo[this.posHistoryLen] = this.hash[1];
this.posHistoryLen++;

// In repetition check:
isRepetition() {
    const h0 = this.hash[0], h1 = this.hash[1];
    // Only check back to last irreversible move (halfmove clock)
    const start = Math.max(0, this.posHistoryLen - this.halfmove);
    for (let i = this.posHistoryLen - 2; i >= start; i -= 2) {
        if (this.posHistoryHi[i] === h0 && this.posHistoryLo[i] === h1) return true;
    }
    return false;
}
```

The `i -= 2` skip is valid because same-side positions only occur on even plies.

### 8. **Consolidate null-move into proper makeNullMove/unmakeNullMove**

```js
// Current: 25 lines of manual state save/restore duplicating makeMove logic
// Replace with:
makeNullMove() {
    this.posHistoryHi[this.posHistoryLen] = this.hash[0];
    this.posHistoryLo[this.posHistoryLen++] = this.hash[1];
    this.stateStack.push({ castling: this.castling, epSquare: this.epSquare,
        halfmove: this.halfmove, hash: [this.hash[0], this.hash[1]] });
    if (this.epSquare >= 0) {
        const ef = sqFile(this.epSquare) * 2;
        zobXor(this.hash, [ZOBRIST.epKeys[ef], ZOBRIST.epKeys[ef + 1]]);
    }
    this.epSquare = -1;
    this.side = -this.side;
    zobXor(this.hash, ZOBRIST.sideKey);
    this.halfmove++;
}

unmakeNullMove() {
    this.side = -this.side;
    this.posHistoryLen--;
    const st = this.stateStack.pop();
    Object.assign(this, st); // or assign fields individually
}
```

Cuts ~40 lines from negamax, eliminates bugs from incomplete state restore.
